Getting Started

Install the library
npm install @timalexlafferty/phoebe-charts
Import a component
import { LineChartComponent } from '@timalexlafferty/phoebe-charts';

@Component({
  imports: [LineChartComponent],
  template: `
    <phoebe-line-chart [data]="chartData"></phoebe-line-chart>
  `
})
export class MyComponent { }
